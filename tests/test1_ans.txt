; ModuleID = 'module'
source_filename = "module"

@a = global i32 10
@b = global i32 5
@c = global i32 0

define i32 @main() {
mainEntry:
  %lVal = load i32, i32* @a, align 4
  %lVal1 = load i32, i32* @b, align 4
  %gtCondVal = icmp sgt i32 %lVal, %lVal1
  %zExtCondVal = zext i1 %gtCondVal to i32
  %lVal2 = load i32, i32* @c, align 4
  %gtCondVal3 = icmp sgt i32 %zExtCondVal, %lVal2
  %zExtCondVal4 = zext i1 %gtCondVal3 to i32
  %condition = icmp ne i32 %zExtCondVal4, 0
  br i1 %condition, label %if_true, label %if_false

if_true:                                          ; preds = %mainEntry
  store i32 2, i32* @a, align 4
  br label %if_exit

if_false:                                         ; preds = %mainEntry
  store i32 20, i32* @a, align 4
  br label %if_exit

if_exit:                                          ; preds = %if_false, %if_true
  %lVal5 = load i32, i32* @a, align 4
  ret i32 %lVal5
}
